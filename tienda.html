async function cargarMasVendidos() {

`${SUPABASE_URL}/rest/v1/pedidos?select=items&order=created_at.desc&limit=50`,

);

const pedidos = await res.json();
const freq = {};
pedidos.forEach(p => {
(p.items || []).forEach(item => {
freq[item.id] = (freq[item.id] || 0) + item.qty;
});
});
return Object.entries(freq)
.sort((a, b) => b[1] - a[1])
.slice(0, 6)
.map(([id]) => Number(id));
} catch { return []; }
}

/* ═══════════════════════════════════════════════════════════

function addToCart(id) {
const p = allProducts.find(x => x.id === id);
if (!p) return;
const existing = cart.find(x => x.id === id);
if (existing) { existing.qty += 1; }
else { cart.push({ id: p.id, nombre: p.nombre, precio: p.precio, imagen_url: p.imagen_url || '', emoji: getEmoji(p), qty: 1 }); }
const qty = cart.find(x => x.id === id).qty;
renderQtyCtrl(id, qty);
syncBadge();
renderCartItems();
saveCart();
}

function changeQty(id, delta) {
const item = cart.find(x => x.id === id);
if (!item) return;
item.qty = Math.max(0, Math.min(99, item.qty + delta));
if (item.qty === 0) {
cart = cart.filter(x => x.id !== id);
renderAddBtn(id);
} else {
renderQtyCtrl(id, item.qty);
}
syncBadge();
renderCartItems();
saveCart();
}

function removeFromCart(id) {
cart = cart.filter(x => x.id !== id);
renderAddBtn(id);
syncBadge();
renderCartItems();
saveCart();
}

function clearCart() {
cart.forEach(item => renderAddBtn(item.id));
cart = [];
saveCart();
syncBadge();
renderCartItems();
}
